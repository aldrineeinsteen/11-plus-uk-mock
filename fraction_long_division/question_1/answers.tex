\section{Fractions Arithmetic}
\begin{enumerate}
\item $\frac{1}{4} + \frac{3}{4} = 1$.
\item $\frac{4}{5} - \frac{3}{5} = \frac{1}{5}$.
\item $\frac{2}{3} \times \frac{3}{4} = \frac{1}{2}$.
\item $\frac{3}{4} \div \frac{2}{3} = \frac{9}{8}$ or $1\frac{1}{8}$.
\item The fraction $\frac{8}{12}$ simplifies to $\frac{2}{3}$.
\item $\frac{1}{3} + \frac{2}{3} = 1$.
\item $\frac{5}{7} - \frac{2}{7} = \frac{3}{7}$.
\item $\frac{3}{5} \times \frac{4}{5} = \frac{12}{25}$.
\item $\frac{5}{6} \div \frac{3}{4} = \frac{10}{9}$ or $1\frac{1}{9}$.
\item The fraction $\frac{9}{15}$ simplifies to $\frac{3}{5}$.
\end{enumerate}
